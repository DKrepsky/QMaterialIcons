86 // Returns an icon pixmap.
87 QPixmap QMaterialIcons::getPixmap(mi::icon icon, QColor color, int size) {
88  Q_ASSERT_X(NULL != m_instance, "QMaterialIcons library",
89  "Library not loaded.");
90 
91  // Check if resources are loaded.
92  if (NULL == m_instance) {
93  return QPixmap();
94  }
95 
96  // Create a new pixmap with size x size pixels and transparent background.
97  QPixmap pix(size, size);
98  pix.fill(Qt::transparent);
99 
100  // Set painting options.
101  QPainter painter(&pix);
102  QFont painterFont = m_instance->getFont();
103  painterFont.setPixelSize(size);
104  painter.setFont(painterFont);
105  painter.setPen(color);
106 
107  // Draw icon.
108  painter.drawText(0, size, QChar(icon));
109  return pix;
110 }

112 // Returns a QIcon icon.
113 QIcon QMaterialIcons::getIcon(mi::icon icon, QColor color, int size) {
114 
115  // Sanity check is skipped because getPixmap will return an empty pixmap if
116  // something is wrong.
117  return QIcon(getPixmap(icon, color, size));

120 // Class constructor.
121 QMaterialIcons::QMaterialIcons() {
122  // Load font resource. This is necessary in libraries.
123  Q_INIT_RESOURCE(font);
124 
125  // Load font into Qt.
126  int id = QFontDatabase::addApplicationFont(":/font/materialicons.ttf");
127 
128  // Load successful?
129  if (id != -1) {
130  QString family = QFontDatabase::applicationFontFamilies(id).at(0);
131  m_font = QFont(family);
132  } else {
133  qDebug() << "Error loading QMaterialIcons font.";
134  }
135 }
